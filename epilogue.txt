function runScripts() {
    var scripts = window.document.getElementsByTagName('script');
    
    for (var i = 0; i < scripts.length; i++) {
        var script = scripts[i];
        if (script.type != 'text/smalltalk')
            continue;
            
        jpyscript.execute(script.innerHTML, true);
    }
